import numpy as np
import matplotlib.pyplot as plt

# --- –ü–∞—Ä–∞–º–µ—Ç—Ä—ã –º–æ–¥–µ–ª–∏ ---
E_b = 15.0          # –±–∞–∑–æ–≤–∞—è –º–æ—â–Ω–æ—Å—Ç—å –º–æ–∑–≥–∞, –í—Ç
E_s_max = 10.0      # –º–∞–∫—Å–∏–º–∞–ª—å–Ω–∞—è –¥–æ–ø–æ–ª–Ω–∏—Ç–µ–ª—å–Ω–∞—è –º–æ—â–Ω–æ—Å—Ç—å, –í—Ç (—Å—Ç—Ä–µ—Å—Å)
I = 15.0            # –±–∞–∑–æ–≤—ã–π –º–µ—Ç–∞–±–æ–ª–∏—á–µ—Å–∫–∏–π –ø—Ä–∏—Ç–æ–∫ —ç–Ω–µ—Ä–≥–∏–∏, –í—Ç
k_loss = 0.01       # –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç –ø–æ—Ç–µ—Ä—å —ç–Ω–µ—Ä–≥–∏–∏, 1/—Å
R_target = 6000.0   # —Ü–µ–ª–µ–≤–æ–π –∑–∞–ø–∞—Å —ç–Ω–µ—Ä–≥–∏–∏, –î–∂
R_min = 2000.0      # –∫—Ä–∏—Ç–∏—á–µ—Å–∫–∏–π –∑–∞–ø–∞—Å —ç–Ω–µ—Ä–≥–∏–∏, –î–∂
S_max = 8.0         # –º–∞–∫—Å–∏–º–∞–ª—å–Ω–∞—è –ø–æ–¥–ø–∏—Ç–∫–∞ (–æ—Ç —ç–º—É–ª—è—Ç–æ—Ä–∞), –í—Ç
k_p = 0.5           # –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç –ø—Ä–æ–ø–æ—Ä—Ü–∏–æ–Ω–∞–ª—å–Ω–æ–≥–æ —É–ø—Ä–∞–≤–ª–µ–Ω–∏—è –ø–æ–¥–ø–∏—Ç–∫–æ–π
tau_S = 5.0         # –≤—Ä–µ–º—è —Å–≥–ª–∞–∂–∏–≤–∞–Ω–∏—è –ø–æ–¥–ø–∏—Ç–∫–∏, —Å

dt = 0.5            # —à–∞–≥ –∏–Ω—Ç–µ–≥—Ä–∏—Ä–æ–≤–∞–Ω–∏—è, —Å

# --- –§—É–Ω–∫—Ü–∏—è –ø—Ä–æ–∏–∑–≤–æ–¥–∏—Ç–µ–ª—å–Ω–æ—Å—Ç–∏ ---
def performance(R):
    x = R / R_target
    alpha = R_min / R_target
    if x <= alpha:
        return 0.0
    elif x < 1.0:
        return (x - alpha) / (1 - alpha)
    else:
        return 1.0

# --- –§—É–Ω–∫—Ü–∏—è –ø–æ–¥–ø–∏—Ç–∫–∏ –ø–æ –∑–∞–ø—Ä–æ—Å—É q(t) ---
def compute_S(q, R, S_prev, dt):
    # –ü—Ä–æ–ø–æ—Ä—Ü–∏–æ–Ω–∞–ª—å–Ω–æ–µ —É–ø—Ä–∞–≤–ª–µ–Ω–∏–µ –ø–æ–¥–ø–∏—Ç–∫–æ–π —Å –æ–≥—Ä–∞–Ω–∏—á–µ–Ω–∏—è–º–∏ –∏ —Å–≥–ª–∞–∂–∏–≤–∞–Ω–∏–µ–º
    S_raw = q * np.clip(k_p * (R_target - R), 0, S_max)
    # –≠–∫—Å–ø–æ–Ω–µ–Ω—Ü–∏–∞–ª—å–Ω–æ–µ —Å–≥–ª–∞–∂–∏–≤–∞–Ω–∏–µ –ø–æ–¥–ø–∏—Ç–∫–∏
    S = S_prev + (dt / tau_S) * (S_raw - S_prev)
    return S

# --- –û—Å–Ω–æ–≤–Ω–∞—è —Å–∏–º—É–ª—è—Ü–∏—è ---
def simulate():

    P = np.zeros(steps)      # –ø—Ä–æ–∏–∑–≤–æ–¥–∏—Ç–µ–ª—å–Ω–æ—Å—Ç—å
    q = np.zeros(steps)      # –∑–∞–ø—Ä–æ—Å –Ω–∞ –ø–æ–º–æ—â—å (0 –∏–ª–∏ 1)

    stress_start = int(300 / dt)
    stress_end = int(900 / dt)
    
    for t in range(1, steps):
        if stress_start <= t <= stress_end:
            E_s[t] = 10.0
        else:
            E_s[t] = 0.0
        
        if R[t-1] < (R_min + 500):
            q[t:t+int(60/dt)] = 1
        
        S_prev = S[t-1]

        E_total = E_b + E_s[t]
        
        dRdt = I + S[t] - E_total - k_loss * R[t-1]

        P[t] = performance(R[t])

    plt.legend()
    plt.grid()
